Lện if else if else trong js

    Cú pháp lệnh if else if else:
    if(điều kiện 1){
        được thực hiện khi điều kiện 1 == true
    } else if(điều kiện 2){
        được thực hiện khi điều kiện 2 == true
    } else if(điều kiện 3){
        được thực hiện khi điều kiện 3 == true
    } else if(điều kiện 4){
        được thực hiện khi điều kiện 4 == true
    } else(điều kiện 5){
        thực hiện khi các điều kiện 1, 2, 3, 4 == false
    }
